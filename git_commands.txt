# Додавання та коміт змін
git add file.txt        # додати файл до стейджу
git commit -m "Commit message" # коміт з повідомленням

# Перегляд статусу
git status              # перегляд поточного статусу репозиторію

# Синхронізація з віддаленим репозиторієм
git pull                # оновлення локальної копії з віддаленого репозиторію
git push                # відправка змін на віддалений репозиторій
